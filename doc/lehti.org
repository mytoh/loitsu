
*  lehti
   
** commands
   
*** install
    install package
#+BEGIN_SRC shell
lehti install napa
#+END_SRC
    
*** deinstall
    deinstall package
#+BEGIN_SRC shell
lehti deinstall napa
#+END_SRC
    
*** reinstall
    reintall package
#+BEGIN_SRC shell
lehti reinstall napa
#+END_SRC
    
*** environment
    display current environment
#+BEGIN_SRC shell
lehti environment
#+END_SRC
    
*** specification
    display package information
#+BEGIN_SRC shel
lehti specification napa
#+END_SRC
    
*** clean
    clean up files
#+BEGIN_SRC shell
lehti clean
#+END_SRC
    
    clean up package fiels
#+BEGIN_SRC shell
lehti clean napa
#+END_SRC
    
*** contents
    display package files
#+BEGIN_SRC shell
lehti contents napa
#+END_SRC
    
*** list
    display installed packages
#+BEGIN_SRC shell
lehti list
#+END_SRC
